#include<stdio.h>
#include<string.h>
#define PATLEN 30
#define TEXTLEN 100

int main(void)
{
	char text[TEXTLEN];
	int textlen;
	char pattern[PATLEN];
	int patlen;
	printf("Enter text: ");
	fgets(text,TEXTLEN,stdin);
	printf("Enter pattern: ");
	fgets(pattern,PATLEN,stdin);

	textlen=strlen(text);
	patlen=strlen(pattern);
	
	int i,j,x;

	for(i=0;i<textlen-patlen+1;i++)
	{
		x=i;
		
		for(j=0;j<patlen;j++)
		{
			if (pattern[j]==text[x])
				{x++;}
			else
				{break;}
			
			if (x-i==patlen-1)
				{printf("Occurence at index %d\n",i);}
		}
	}
	


	return(0);
}
