def hash_fxn(string,prime_no):
	hash_value=0	
	for index, character in enumerate(string):
		hash_value+=ord(str(character))*(prime_no**index)		
	return hash_value

	test=((last_hash-ord(str(old_character)))/prime)+(ord(str(new_character))*prime**(length-1))
