text = raw_input("Enter text: ")
pattern = raw_input("Enter pattern: ")

n= len(text)
m= len(pattern)

vector= [1 for i in range(m+1)]
vector[0]= 0

for x in range(1,n+1):
	vector[0]= 0
	y= m
	while(y>0):
		vector[y]=int(vector[y-1] or (not(text[x-1]==pattern[y-1])))
		y-=1

	if (vector[m]==0):
		print x-m
